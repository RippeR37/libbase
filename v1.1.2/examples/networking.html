Networking example

Below you can find source code of a networking example that showcases usage of the libbase functionality for performing network requests. This example is located in the /examples/networking/ directory in the repository and - if enabled - will be built along the libbase library itself.

CMakeLists.txt

 2
 3target_compile_options(networking
 4  PRIVATE
 5    ${LIBBASE_COMPILE_FLAGS}
 6)

 8target_link_libraries(networking
 9  PRIVATE
10    libbase
11    libbase_net
12)

16    main.cc

 14void LogNetResponse(const base::net::ResourceResponse& response) {
 15  LOG(INFO) << "Result: " << static_cast<int>(response.result);
 16  LOG(INFO) << "HTTP code: " << response.code;
 17  LOG(INFO) << "Final URL: " << response.final_url;
 18  LOG(INFO) << "Downloaded " << response.data.size() << " bytes";
 19  LOG(INFO) << "Latency: " << response.timing_connect.InMilliseconds() << "ms";
 20  LOG(INFO) << "Headers";
 21  for (const auto& [h, v] : response.headers) {
 22    LOG(INFO) << "  " << h << ": " << v;
 23  }
 24  LOG_IF(INFO, !response.data.empty()) << "Content:\n"
 25                                       << response.DataAsStringView();
 26}
 27
 28void NetExampleGet() {
 29  base::RunLoop run_loop{};
 30
 31  // Try to download and signal on finish
 32  base::net::SimpleUrlLoader::DownloadUnbounded(
 33      base::net::ResourceRequest{"https://www.google.com/robots.txt"}
 34          .WithHeadersOnly()
 35          .WithTimeout(base::Seconds(5)),
 36      base::BindOnce([](base::net::ResourceResponse response) {
 37        LogNetResponse(response);
 38      }).Then(run_loop.QuitClosure()));
 39
 40  // Runs all tasks until the quit callback is called
 41  run_loop.Run();
 42}
 43
 44void NetExamplePost() {
 45  base::RunLoop run_loop{};
 46
 47  // Try to download and signal on finish
 48  base::net::SimpleUrlLoader::DownloadUnbounded(
 49      base::net::ResourceRequest{"https://httpbin.org/post"}
 50          .WithHeaders({"Content-Type: application/json"})
 51          .WithPostData("{\"key\": \"value\"}")
 52          .WithTimeout(base::Seconds(5)),
 53      base::BindOnce([](base::net::ResourceResponse response) {
 54        LogNetResponse(response);
 55      }).Then(run_loop.QuitClosure()));
 56
 57  run_loop.Run();
 58}

 60class UrlRequestExampleUser : public base::net::UrlRequest::Client {
 61 public:
 62  UrlRequestExampleUser(base::OnceClosure finished_closure)
 63      : finished_closure_(std::move(finished_closure)), request_(this) {}
 64
 65  void Download(base::net::ResourceRequest request) {
 66    DCHECK_CALLED_ON_VALID_SEQUENCE(sequence_checker_);
 67    request_.Start(std::move(request));
 68  }
 69
 70  void OnResponseStarted(const base::net::UrlRequest* request,
 71                         int code,
 72                         std::string final_url,
 73                         std::map<std::string, std::string> headers) override {
 74    DCHECK_CALLED_ON_VALID_SEQUENCE(sequence_checker_);
 75    DCHECK_EQ(request, &request_);
 76
 77    LOG(INFO) << __FUNCTION__ << "() code: " << code;
 78    LOG(INFO) << __FUNCTION__ << "() final_url: " << final_url;
 79    for (const auto& [k, v] : headers) {
 80      LOG(INFO) << __FUNCTION__ << "() header[" << k << "]: " << v;
 81    }
 82  }
 83
 84  void OnWriteData(const base::net::UrlRequest* request,
 85                   std::vector<uint8_t> chunk) override {
 86    DCHECK_CALLED_ON_VALID_SEQUENCE(sequence_checker_);
 87    DCHECK_EQ(request, &request_);
 88
 89    LOG(INFO) << __FUNCTION__ << "() received bytes: " << chunk.size();
 90  }
 91
 92  void OnRequestFinished(const base::net::UrlRequest* request,
 93                         base::net::Result result) override {
 94    DCHECK_CALLED_ON_VALID_SEQUENCE(sequence_checker_);
 95    DCHECK_EQ(request, &request_);
 96
 97    LOG(INFO) << __FUNCTION__ << "() result: " << static_cast<int>(result);
 98    std::move(finished_closure_).Run();
 99  }
100
101 private:
102  SEQUENCE_CHECKER(sequence_checker_);
103  base::OnceClosure finished_closure_;
104  base::net::UrlRequest request_;
105};
106
107void NetExampleUrlRequest() {
108  base::RunLoop run_loop;
109
110  UrlRequestExampleUser url_request_user{run_loop.QuitClosure()};
111  url_request_user.Download(
112      base::net::ResourceRequest{"https://www.google.com/robots.txt"}
113          .WithTimeout(base::Seconds(5)));
114
115  run_loop.Run();
116}
117
118int main(int argc, char* argv[]) {
119  base::Initialize(argc, argv, base::InitOptions{});
120  base::net::Initialize(base::net::InitOptions{});
121
122  const auto timer = base::ElapsedTimer{};
123
124  NetExampleGet();
125  NetExamplePost();
126  NetExampleUrlRequest();
127
128  LOG(INFO) << "Example finished in " << timer.Elapsed().InMillisecondsF()
129            << "ms";
130
131  base::net::Deinitialize();
132  base::Deinitialize();
133  return 0;
134}
